Admin username and password

Prompt user with username. If it's "Admin", prompt with password. If it's "TheMaster", greet. If password is wrong, display message. If it's not admin, display message. If canceled, display cancel message.

let userName = prompt ("User name:", "");
if (userName === null) {
    alert ("Canceled");
} else if (userName === "Admin") {
    let userPass = prompt ("Password:", "");
    if (userPass === null) {
        alert ("Canceled");
    } else if (userPass === "TheMaster") {
        alert (`Greetings, ${userName}!`);
    } else {
        alert ("Wrong password");
    }
} else {
    alert ("I don't know you.");
}
