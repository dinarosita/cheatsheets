Admin login

Here's a special button for admin login. Only accepts Admin (not case sensitive) at login. Password is "TheMaster" (case sensitive).

<input type="button" id="loginButton" value="Login">
<script>
function loginFunction() {
let userName1 = prompt("Who are you?");
if (userName1 === null) {
alert ("Nevermind...");
} else if (userName1.toUpperCase() === "ADMIN") {
let userPass1 = prompt ("Enter your password: ");
if (userPass1 === null) {
alert ("Sigh...");
} else if (userPass1 === "TheMaster") {
alert ("Hello, Admin!");
} else {
alert ("No no no no no... not today!");
}
} else {
alert ("Sorry, we don't know you.");
}
}
loginButton.onclick = loginFunction;
</script>
